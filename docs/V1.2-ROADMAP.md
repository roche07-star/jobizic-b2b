CREATE TABLE contact_requests (
  id UUID PRIMARY KEY DEFAULT uuid_generate_v4(),
  candidate_id UUID NOT NULL REFERENCES candidates(id) ON DELETE CASCADE,
  requester_id UUID NOT NULL REFERENCES auth.users(id),
  requester_email VARCHAR(255) NOT NULL,
  status VARCHAR(20) NOT NULL DEFAULT 'pending', -- pending, approved, rejected
  created_at TIMESTAMP DEFAULT NOW(),
  responded_at TIMESTAMP,
  responded_by UUID REFERENCES auth.users(id),
  UNIQUE(candidate_id, requester_id)
);

CREATE INDEX idx_contact_requests_candidate ON contact_requests(candidate_id);
CREATE INDEX idx_contact_requests_requester ON contact_requests(requester_id);
CREATE INDEX idx_contact_requests_status ON contact_requests(status);

2. **권한 로직** (lib/contact-permissions.ts)
```typescript
export interface ContactPermission {
  canView: boolean
  reason: 'owner' | 'creator' | 'approved' | 'none'
}

export function checkContactPermission(
  candidate: Candidate,
  profile: Profile,
  approvedRequests: string[]
): ContactPermission {
  // Owner/Admin: 모든 연락처 볼 수 있음
  if (profile.role === 'owner' || profile.role === 'admin') {
    return { canView: true, reason: 'owner' }
  }

  // 본인이 등록한 후보자
  if (candidate.created_by === profile.email) {
    return { canView: true, reason: 'creator' }
  }

  // 승인된 요청
  if (approvedRequests.includes(candidate.id)) {
    return { canView: true, reason: 'approved' }
  }

  return { canView: false, reason: 'none' }
}

export function maskContact(value: string, type: 'email' | 'phone'): string {
  if (!value) return '***'
  
  if (type === 'email') {
    const [local, domain] = value.split('@')
    if (!domain) return '***'
    return `${local.slice(0, 2)}***@${domain}`
  }
  
  if (type === 'phone') {
    // 010-1234-5678 → 010-****-5678
    const cleaned = value.replace(/[^0-9]/g, '')
    if (cleaned.length === 11) {
      return `${cleaned.slice(0, 3)}-****-${cleaned.slice(7)}`
    }
    return '***-****-***'
  }
  
  return '***'
}
```

export async function POST(req: NextRequest) {
  const { candidate_id } = await req.json()
  const profile = await getProfile()

  // 중복 요청 확인
  const { data: existing } = await supabase
    .from('contact_requests')
    .select('*')
    .eq('candidate_id', candidate_id)
    .eq('requester_id', profile.id)
    .single()

  if (existing) {
    return NextResponse.json({ 
      error: '이미 요청하셨습니다.',
      status: existing.status 
    }, { status: 400 })
  }

  // 요청 생성
  const { data: request } = await supabase
    .from('contact_requests')
    .insert({
      candidate_id,
      requester_id: profile.id,
      requester_email: profile.email,
    })
    .select()
    .single()

  // 후보자 등록자에게 알림
  const { data: candidate } = await supabase
    .from('candidates')
    .select('name, created_by')
    .eq('id', candidate_id)
    .single()

  const { data: owner } = await supabase
    .from('profiles')
    .select('id')
    .eq('email', candidate.created_by)
    .single()

  await createNotification({
    userId: owner.id,
    type: 'contact_request',
    title: '연락처 열람 요청',
    message: `${profile.full_name || profile.email}님이 "${candidate.name}" 후보자의 연락처를 요청했습니다.`,
    relatedId: request.id,
    relatedType: 'contact_request',
    actionUrl: '/contact-requests',
  })

  return NextResponse.json({ success: true, request })
}

**PATCH /api/contact-requests/[id]** (승인/거부)
```typescript
export async function PATCH(req: NextRequest, { params }: { params: Promise<{ id: string }> }) {
  const { id } = await params
  const { status } = await req.json() // 'approved' or 'rejected'
  const profile = await getProfile()

  // 요청 조회
  const { data: request } = await supabase
    .from('contact_requests')
    .select(`
      *,
      candidates(created_by)
    `)
    .eq('id', id)
    .single()

  // 권한 확인: 후보자 등록자 또는 Owner만 승인/거부 가능
  if (request.candidates.created_by !== profile.email && 
      profile.role !== 'owner' && 
      profile.role !== 'admin') {
    return NextResponse.json({ error: '권한이 없습니다.' }, { status: 403 })
  }

  // 상태 업데이트
  await supabase
    .from('contact_requests')
    .update({
      status,
      responded_at: new Date().toISOString(),
      responded_by: profile.id,
    })
    .eq('id', id)

  // 요청자에게 알림
  if (status === 'approved') {
    await createNotification({
      userId: request.requester_id,
      type: 'contact_approved',
      title: '연락처 열람 승인',
      message: `"${request.candidates.name}" 후보자의 연락처가 승인되었습니다.`,
      actionUrl: '/candidates',
    })
  }

  return NextResponse.json({ success: true })
}
```

**GET /api/contact-requests** (내 요청 목록)
```typescript
export async function GET(req: NextRequest) {
  const profile = await getProfile()
  const type = req.nextUrl.searchParams.get('type') // 'sent' or 'received'

  let query = supabase
    .from('contact_requests')
    .select(`
      *,
      candidates(id, name, current_company, current_position),
      profiles!contact_requests_requester_id_fkey(full_name, email)
    `)

  if (type === 'sent') {
    query = query.eq('requester_id', profile.id)
  } else if (type === 'received') {
    // 내가 등록한 후보자에 대한 요청
    query = query.filter('candidates.created_by', 'eq', profile.email)
  }

  const { data: requests } = await query.order('created_at', { ascending: false })

  return NextResponse.json({ requests })
}
```

'use client'

import { useState, useEffect } from 'react'

export default function ContactRequestsPage() {
  const [sentRequests, setSentRequests] = useState([])
  const [receivedRequests, setReceivedRequests] = useState([])
  const [tab, setTab] = useState<'sent' | 'received'>('received')

  useEffect(() => {
    loadRequests()
  }, [])

  async function loadRequests() {
    const [sent, received] = await Promise.all([
      fetch('/api/contact-requests?type=sent').then(r => r.json()),
      fetch('/api/contact-requests?type=received').then(r => r.json()),
    ])
    setSentRequests(sent.requests || [])
    setReceivedRequests(received.requests || [])
  }

  async function handleResponse(id: string, status: 'approved' | 'rejected') {
    await fetch(`/api/contact-requests/${id}`, {
      method: 'PATCH',
      headers: { 'Content-Type': 'application/json' },
      body: JSON.stringify({ status }),
    })
    loadRequests()
  }

  return (
    <main className="page">
      <div className="page-header">
        <div className="page-title">연락처 요청 관리</div>
      </div>

      <div className="filter-bar">
        <button 
          className={`filter-btn${tab === 'received' ? ' active' : ''}`}
          onClick={() => setTab('received')}
        >
          받은 요청 ({receivedRequests.filter(r => r.status === 'pending').length})
        </button>
        <button 
          className={`filter-btn${tab === 'sent' ? ' active' : ''}`}
          onClick={() => setTab('sent')}
        >
          보낸 요청 ({sentRequests.length})
        </button>
      </div>

      {tab === 'received' && (
        <div className="card">
          {receivedRequests.map(req => (
            <div key={req.id} style={{ padding: 16, borderBottom: '1px solid var(--border)' }}>
              <div style={{ display: 'flex', justifyContent: 'space-between', alignItems: 'center' }}>
                <div>
                  <div style={{ fontWeight: 600 }}>{req.candidates.name}</div>
                  <div style={{ fontSize: 12, color: 'var(--muted2)' }}>
                    요청자: {req.profiles.full_name || req.profiles.email}
                  </div>
                  <div style={{ fontSize: 11, color: 'var(--muted2)' }}>
                    {new Date(req.created_at).toLocaleString('ko-KR')}
                  </div>
                </div>
                <div style={{ display: 'flex', gap: 8 }}>
                  {req.status === 'pending' && (
                    <>
                      <button 
                        className="btn btn-success btn-sm"
                        onClick={() => handleResponse(req.id, 'approved')}
                      >
                        승인
                      </button>
                      <button 
                        className="btn btn-danger btn-sm"
                        onClick={() => handleResponse(req.id, 'rejected')}
                      >
                        거부
                      </button>
                    </>
                  )}
                  {req.status !== 'pending' && (
                    <span className={`badge badge-${req.status}`}>
                      {req.status === 'approved' ? '승인됨' : '거부됨'}
                    </span>
                  )}
                </div>
              </div>
            </div>
          ))}
        </div>
      )}

      {tab === 'sent' && (
        <div className="card">
          {sentRequests.map(req => (
            <div key={req.id} style={{ padding: 16, borderBottom: '1px solid var(--border)' }}>
              <div style={{ display: 'flex', justifyContent: 'space-between', alignItems: 'center' }}>
                <div>
                  <div style={{ fontWeight: 600 }}>{req.candidates.name}</div>
                  <div style={{ fontSize: 12, color: 'var(--muted2)' }}>
                    {req.candidates.current_company} · {req.candidates.current_position}
                  </div>
                </div>
                <span className={`badge badge-${req.status}`}>
                  {req.status === 'pending' && '대기중'}
                  {req.status === 'approved' && '승인됨'}
                  {req.status === 'rejected' && '거부됨'}
                </span>
              </div>
            </div>
          ))}
        </div>
      )}
    </main>
  )
}
